# 1.动态路由(页面刷新数据不丢失)

```
methods： {
    goTo(id) {
        <!-- $router.push 实现跳转 -->
        this.$router.push({
            path: `/home/detail/${id}`
        }
        )
    }
}

路由配置

{
    path: `/home/detail/:id`,
    name: 'detail',
    component: detail
}


页面参数接收

this.$route.params.id
```
# 2.name匹配，params传递

```
methods： {
    goTo(id) {
        <!-- $router.push 实现跳转 -->
        this.$router.push({
            name: 'detail',
            params: {
                id: id
            }
        }
        )
    }
}

路由配置

{
    path: `/home/detail`,
    name: 'detail',
    component: detail
}


页面参数接收

this.$route.params.id


```

# 3.path路径匹配，query传参

```
methods： {
    goTo(id) {
        this.$router.push({
            path: '/home/detail',
            query: {
                id: id
            }
        }
        )
    }
}

路由配置

{
    path: `/home/detail`,
    name: 'detail',
    component: detail
}


页面参数接收

this.$route.query.id
```


- query=====>path
- params====>name
